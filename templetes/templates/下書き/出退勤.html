import tkinter as tk from tkinter import messagebox from datetime import datetime def register_attendance(): current_time = datetime.now().strftime('%Y-%m-%d %H:%M:%S') status = "出勤" if attendance_status.get() else "退勤" message = f"{status}時間: {current_time}" messagebox.showinfo("出退勤入力", message) # Tkinterウィンドウを作成 root = tk.Tk() root.title("出退勤入力画面") # ユーザー名入力フレーム user_frame = tk.Frame(root) user_frame.pack(pady=10) tk.Label(user_frame, text="ユーザー名:").pack(side=tk.LEFT) user_name_entry = tk.Entry(user_frame) user_name_entry.pack(side=tk.LEFT) # チェックボックスフレーム status_frame = tk.Frame(root) status_frame.pack(pady=10) attendance_status = tk.BooleanVar() attendance_status.set(False) # チェックを外す状態に初期化 tk.Checkbutton(status_frame, text="出勤", variable=attendance_status).pack(side=tk.LEFT) # 登録ボタン register_button = tk.Button(root, text="登録", command=register_attendance) register_button.pack(pady=10) # ウィンドウをループして表示 root.mainloop()
